题解 luogu P1829 【[国家集训队]Crash的数字表格 / JZPTAB】

发布于 2020-06-26 | 分类于 莫反 、 数学/数论 、 题解 | 5分钟 | 775字数

一句话题意：求

∑i=1n∑j=1mlcm⁡(i,j) mod 20101009\sum_{i=1}^n\sum_{j=1}^m\operatorname{lcm}(i,j)\bmod 20101009 i=1∑n​j=1∑m​lcm(i,j)mod20101009

数据范围：1≤n,m≤1071\le n,m\le 10^71≤n,m≤107。

莫反好题。

Solution

（默认 n≤mn\le mn≤m）

∑i=1n∑j=1mlcm⁡(i,j)=∑i=1n∑j=1mijgcd⁡(i,j)=∑i=1n∑j=1m∑d∣i∧d∣j∧gcd⁡(id,jd)=1ijd=∑d=1nd∑i=1⌊nd⌋∑j=1⌊md⌋[gcd⁡(i,j)=1]⋅ij\begin{aligned} &\sum_{i=1}^n\sum_{j=1}^m\operatorname{lcm}(i,j)\\ =&\sum_{i=1}^n\sum_{j=1}^m\frac{ij}{\gcd(i,j)}\\ =&\sum_{i=1}^n\sum_{j=1}^m\sum_{d|i\land d|j\land \gcd\left(\frac id,\frac jd\right)=1}\frac{ij}d\\ =&\sum_{d=1}^n d\sum_{i=1}^{\left\lfloor\frac nd\right\rfloor}\sum_{j=1}^{\left\lfloor\frac md\right\rfloor}[\gcd(i,j)=1]\cdot ij\\ \end{aligned} ===​i=1∑n​j=1∑m​lcm(i,j)i=1∑n​j=1∑m​gcd(i,j)ij​i=1∑n​j=1∑m​d∣i∧d∣j∧gcd(di​,dj​)=1∑​dij​d=1∑n​di=1∑⌊dn​⌋​j=1∑⌊dm​⌋​[gcd(i,j)=1]⋅ij​

令

sum⁡(n,m)=∑i=1n∑j=1m[gcd⁡(i,j)=1]⋅ij\operatorname{sum}(n,m)=\sum_{i=1}^n\sum_{j=1}^m [\gcd(i,j)=1]\cdot ij sum(n,m)=i=1∑n​j=1∑m​[gcd(i,j)=1]⋅ij

则原式等于

∑d=1nd⋅sum⁡(⌊nd⌋,⌊md⌋)\sum_{d=1}^n d\cdot \operatorname{sum}\left(\left\lfloor\frac nd\right\rfloor,\left\lfloor\frac md\right\rfloor\right) d=1∑n​d⋅sum(⌊dn​⌋,⌊dm​⌋)

化简之：

=∑d=1n∑d∣in∑d∣jmμ(d)⋅ij=∑d=1nμ(d)⋅d2⋅∑i=1⌊nd⌋∑j=1⌊md⌋ij=∑d=1nμ(d)⋅d2⋅g(⌊nd⌋,⌊md⌋)\begin{aligned} =&\sum_{d=1}^n\sum_{d|i}^n\sum_{d|j}^m\mu(d)\cdot ij\\ =&\sum_{d=1}^n\mu(d)\cdot d^2\cdot \sum_{i=1}^{\left\lfloor\frac nd\right\rfloor}\sum_{j=1}^{\left\lfloor\frac md\right\rfloor}ij\\ =&\sum_{d=1}^n\mu(d)\cdot d^2\cdot g\left(\left\lfloor\frac nd\right\rfloor,\left\lfloor\frac md\right\rfloor\right) \end{aligned} ===​d=1∑n​d∣i∑n​d∣j∑m​μ(d)⋅ijd=1∑n​μ(d)⋅d2⋅i=1∑⌊dn​⌋​j=1∑⌊dm​⌋​ijd=1∑n​μ(d)⋅d2⋅g(⌊dn​⌋,⌊dm​⌋)​

其中

g(n,m)=∑i=1n∑j=1mij=n(n+1)2⋅m(m+1)2g(n,m)=\sum_{i=1}^n\sum_{j=1}^m ij=\frac{n(n+1)}2\cdot \frac{m(m+1)}2 g(n,m)=i=1∑n​j=1∑m​ij=2n(n+1)​⋅2m(m+1)​

两次数论分块即可。

Code

#include<cstdio>
#include<algorithm>

const int N=1e7;
const int P=20101009;

int n,m,prime[N+10],totp,mu[N+10],sum[N+10];
bool notPrime[N+10];

inline void init()
{
	mu[1]=1;
	for(int i=2;i<=N;i++)
	{
		if(!notPrime[i])prime[++totp]=i,mu[i]=-1;
		for(int j=1;j<=totp&&i*prime[j]<=N;j++)
		{
			notPrime[i*prime[j]]=1;
			if(i%prime[j]==0)
			{
				mu[i*prime[j]]=0;
				break;
			}
			mu[i*prime[j]]=-mu[i];
		}
	}
	for(int i=1;i<=N;i++)
		sum[i]=(sum[i-1]+1LL*i*i%P*(mu[i]+P))%P;
}

inline int g(int x,int y)
{
	return (1LL*x*(x+1)/2%P)*(1LL*y*(y+1)/2%P)%P;
}

inline int f(int x,int y)
{
	int res=0;
	for(int l=1,r;l<=std::min(x,y);l=r+1)
	{
		r=std::min(x/(x/l),y/(y/l));
		res=(res+1LL*(sum[r]-sum[l-1]+P)*g(x/l,y/l)%P)%P;
	}
	return res;
}

int main()
{
	init();
	scanf("%d%d",&n,&m);
	int ans=0;
	for(int l=1,r;l<=std::min(n,m);l=r+1)
	{
		r=std::min(n/(n/l),m/(m/l));
		ans=(ans+1LL*(r-l+1)*(r+l)/2%P*f(n/l,m/l)%P)%P;
	}
	printf("%d\n",ans);
	return 0;
}
