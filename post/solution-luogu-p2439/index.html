题解 P2439 【[SDOI2005]阶梯教室设备利用】

发布于 2020-07-01 | 分类于 线段树 、 DS 、 DP 、 题解 | 3分钟 | 449字数

这题解法好像很多。

我写的是 DS 优化 DP。

Solution

设 fif_ifi​ 表示前 iii 个演讲能获得的最大演讲时间。

显然，对于一个起始时间为 aaa，终止时间为 bbb 的演讲，有

fb=max⁡{fb,max⁡i=0afi+b−a}f_b=\max\left\{f_b,\max_{i=0}^a f_i+b-a\right\} fb​=max{fb​,i=0maxa​fi​+b−a}

然后 max⁡i=0afi\displaystyle\max_{i=0}^a f_ii=0maxa​fi​ 就能用线段树去求了。

这题很良心qwq，不用离散化。

Code

// for a pair (a, b),
// f[b] = max(f[b], max(f[i]) + b - a) i = 0 to a

#include<cstdio>
#include<algorithm>

const int N=1e5;

struct E
{
	int x,y;
};

bool operator<(const E& x,const E& y)
{
	return x.y<y.y;
}

struct Node
{
	int v;
};

int n,mx,f[N+10];
E a[N+10];
Node t[(N<<2)+10];

#define ls(x) (x<<1)
#define rs(x) (x<<1|1)

inline void pushUp(int i)
{
	t[i].v=std::max(t[ls(i)].v,t[rs(i)].v);
}

void modify(int i,int l,int r,int pos,int v)
{
	if(l==r)
	{
		t[i].v=std::max(t[i].v,v);
		return;
	}
	int mid=(l+r)>>1;
	if(pos<=mid)modify(ls(i),l,mid,pos,v);
	else modify(rs(i),mid+1,r,pos,v);
	pushUp(i);
}

int query(int i,int l,int r,int ql,int qr)
{
	if(ql<=l&&r<=qr)return t[i].v;
	int mid=(l+r)>>1;
	int res=0;
	if(ql<=mid)res=std::max(res,query(ls(i),l,mid,ql,qr));
	if(qr>mid) res=std::max(res,query(rs(i),mid+1,r,ql,qr));
	return res;
}

#undef ls
#undef rs

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d%d",&a[i].x,&a[i].y);
	std::sort(a+1,a+n+1);
	mx=a[n].y;
	for(int i=1;i<=n;i++)
	{
		f[a[i].y]=std::max(f[a[i].y],query(1,0,mx,0,a[i].x)+a[i].y-a[i].x);
		modify(1,0,mx,a[i].y,f[a[i].y]);
	}
	printf("%d\n",query(1,0,mx,0,mx));
	return 0;
}
